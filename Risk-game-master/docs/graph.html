        function continentFrac(node) {
        var allyNum = 0;
        var owner = node.owner;
        var icont = node.continent;
        var contclist = cont[icont];
        _.each(contclist, function(n) {
            if (g.nodes[n].owner == owner) allyNum++;
        })
        return allyNum / contclist.length;
    };
